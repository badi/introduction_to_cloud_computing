    def __init__(self):
        self.vms = {}
        self.status = []
        
    def refresh(self):
        lines = grep (nova("list"), "|").split("\n")[1:-1]
        self.status = []
        
        for line in lines:
            line = line[2:-2]
            (id, name, status, networks) = line.split("|")
            self.vms[id] = {"id": id.strip(),
                       "name":name.strip(),
                       "status":status.strip(),
                       "networks": networks.strip()}

    def get_status(self):
        self.status = []
        for vm in self.vms:
